32pub fn list_devices() -> Vec<AudioDevice> {
33    let host = cpal::default_host();
34    host.input_devices()
35        .map(|devs| {
36            devs.filter_map(|d| {
37                let name = match d.description() {
38                    Ok(desc) => desc,
39                    Err(_) => "unknown".to_string(),
40                };
41                let is_loopback = name.to_lowercase().contains("monitor");
42                Some(AudioDevice {
43                    name,
44                    is_loopback,
45                    device: d,
46                })
47            })
48            .collect()
49        })
50        .unwrap_or_default()
51}

70const FFT_SIZE: usize = 1024;
71const SMOOTH: f32 = 0.3; // lower = smoother / slower response
72
73pub fn start_capture(dev: AudioDevice, shared: SharedAudio) -> anyhow::Result<Stream> {
74    let config = dev.device.default_input_config()?;
75    let sample_rate = config.sample_rate() as f32;
76
77    // Ring buffer: audio thread writes, FFT thread reads
78    let rb = HeapRb::<f32>::new(FFT_SIZE * 4);
79    let (mut prod, mut cons) = rb.split();
80
81    // Spawn FFT analysis thread
82    let shared_clone = shared.clone();
83    std::thread::spawn(move || {
84        let mut planner = FftPlanner::<f32>::new();
85        let fft = planner.plan_fft_forward(FFT_SIZE);
86        let mut scratch = vec![Complex::new(0.0, 0.0); fft.get_inplace_scratch_len()];
87        let mut buf = vec![0.0f32; FFT_SIZE];
88        let mut smooth = BandEnergy::default();
89
90        loop {
91            // Wait until we have a full window
92            let available = cons.occupied_len();
93            if available < FFT_SIZE {
94                std::thread::sleep(std::time::Duration::from_millis(5));
95                continue;
96            }
97            // Drain the oldest samples (skip if we lag behind)
98            let skip = available.saturating_sub(FFT_SIZE);
99            for _ in 0..skip {
100                cons.try_pop();
101            }
102
103            // Read FFT_SIZE samples
104            for s in buf.iter_mut() {
105                *s = cons.try_pop().unwrap_or(0.0);
106            }
107
108            // Apply Hann window
109            let mut spectrum: Vec<Complex<f32>> = buf
110                .iter()
111                .enumerate()
112                .map(|(i, &s)| {
113                    let w = 0.5
114                        * (1.0
115                            - (2.0 * std::f32::consts::PI * i as f32 / (FFT_SIZE - 1) as f32)
116                                .cos());
117                    Complex::new(s * w, 0.0)
118                })
119                .collect();
120
121            fft.process_with_scratch(&mut spectrum, &mut scratch);
122
123            // Convert to magnitude (only first half = positive freqs)
124            let mags: Vec<f32> = spectrum[..FFT_SIZE / 2].iter().map(|c| c.norm()).collect();
125
126            // Bin → Hz: freq = bin * sample_rate / FFT_SIZE
127            let bin_hz = sample_rate / FFT_SIZE as f32;
128
129            let bass = band_energy(&mags, bin_hz, 20.0, 90.0);
130            let mid = band_energy(&mags, bin_hz, 1400.0, 4000.0);
131            let high = band_energy(&mags, bin_hz, 7000.0, 20000.0);
132
133            // Smooth with exponential moving average
134            smooth.bass = smooth.bass * (1.0 - SMOOTH) + bass * SMOOTH;
135            smooth.mid = smooth.mid * (1.0 - SMOOTH) + mid * SMOOTH;
136            smooth.high = smooth.high * (1.0 - SMOOTH) + high * SMOOTH;
137
138            if let Ok(mut state) = shared_clone.lock() {
139                *state = smooth.clone();
140            }
141        }
142    });
143
144    // Build CPAL stream (f32 samples only; convert i16/u16 if needed)
145    let stream = match config.sample_format() {
146        SampleFormat::F32 => build_stream_f32(&dev.device, &config.into(), prod)?,
147        SampleFormat::I16 => build_stream_i16(&dev.device, &config.into(), prod)?,
148        _ => anyhow::bail!("Unsupported sample format"),
149    };
150    stream.play()?;
151    Ok(stream)
152}

154fn band_energy(mags: &[f32], bin_hz: f32, lo: f32, hi: f32) -> f32 {
155    let lo_bin = (lo / bin_hz) as usize;
156    let hi_bin = ((hi / bin_hz) as usize).min(mags.len() - 1);
157    if lo_bin >= hi_bin {
158        return 0.0;
159    }
160    let sum: f32 = mags[lo_bin..=hi_bin].iter().sum();
161    let avg = sum / (hi_bin - lo_bin + 1) as f32;
162    // Soft normalisation — tuned for typical mic levels
163    (avg / 500.0).clamp(0.0, 1.0)
164}

166type RbProd = ringbuf::wrap::caching::Caching<
167    Arc<ringbuf::SharedRb<ringbuf::storage::Heap<f32>>>,
168    true,
169    false,
170>;

172fn build_stream_f32(
173    device: &Device,
174    config: &StreamConfig,
175    mut prod: RbProd,
176) -> anyhow::Result<Stream> {
177    let channels = config.channels as usize;
178    let stream = device.build_input_stream(
179        config,
180        move |data: &[f32], _| {
181            // Mix to mono
182            for frame in data.chunks(channels) {
183                let mono = frame.iter().sum::<f32>() / channels as f32;
184                let _ = prod.try_push(mono);
185            }
186        },
187        |e| eprintln!("Audio stream error: {e}"),
188        None,
189    )?;
190    Ok(stream)
191}

193fn build_stream_i16(
194    device: &Device,
195    config: &StreamConfig,
196    mut prod: RbProd,
197) -> anyhow::Result<Stream> {
198    let channels = config.channels as usize;
199    let stream = device.build_input_stream(
200        config,
201        move |data: &[i16], _| {
202            for frame in data.chunks(channels) {
203                let mono = frame
204                    .iter()
205                    .map(|&s| s as f32 / i16::MAX as f32)
206                    .sum::<f32>()
207                    / channels as f32;
208                let _ = prod.try_push(mono);
209            }
210        },
211        |e| eprintln!("Audio stream error: {e}"),
212        None,
213    )?;
214    Ok(stream)
