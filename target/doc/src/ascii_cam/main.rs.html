108#[derive(Clone, Copy, PartialEq, Debug)]
109pub enum BandTarget {
110    None,
111    Contrast,
112    Resolution,
113    Script,
114    Density,
115}
116
117impl BandTarget {
118    fn name(self) -> &'static str {
119        match self {
120            BandTarget::None       => "—",
121            BandTarget::Contrast   => "Kontrast",
122            BandTarget::Resolution => "Auflösung",
123            BandTarget::Script     => "Schrift",
124            BandTarget::Density    => "Dichte",
125        }
126    }
127
128    fn cycle(self) -> BandTarget {
129        use BandTarget::*;
130        match self { None => Contrast, Contrast => Resolution,
131                     Resolution => Script, Script => Density, Density => None }
132    }
133}
134
135// ── App State ─────────────────────────────────────────────────────────────────
136
137struct State {
138    script: Script,
139    density: f32,      // 0.3–1.0: fraction of charset used
140    inverted: bool,
141    contrast: f32,     // 0.5–4.0
142    scale: u32,        // terminal cols / scale = ascii cols
143
144    mappings: [BandTarget; 3],
145    active_band: usize,
146}
147
148impl Default for State {
149    fn default() -> Self {
150        State {
151            script: Script::Latin,
152            density: 1.0,
153            inverted: false,
154            contrast: 1.0,
155            scale: 2,
156            mappings: [BandTarget::Contrast, BandTarget::Resolution, BandTarget::Script],
157            active_band: 0,
158        }
159    }
160}
161
162impl State {
163    fn charset(&self) -> &'static [char] {
164        let all = self.script.chars();
165        let take = ((all.len() as f32 * self.density) as usize).max(2).min(all.len());
166        &all[..take]
167    }
168
169    fn apply_audio(&mut self, band: &BandEnergy) {
170        let energies = [band.bass, band.mid, band.high];
171        for (i, &e) in energies.iter().enumerate() {
172            match self.mappings[i] {
173                BandTarget::Contrast   => self.contrast = 0.5 + e * 3.5,
174                BandTarget::Resolution => self.scale = (1.0 + e * 5.0).round() as u32,
175                BandTarget::Density    => self.density = (0.3 + e * 0.7).clamp(0.0, 1.0),
176                BandTarget::Script | BandTarget::None => {}
177            }
178        }
179    }
180}

182// ── ASCII conversion ──────────────────────────────────────────────────────────
183
184fn frame_to_ascii(gray: &GrayImage, width: u32, height: u32, state: &State) -> Vec<String> {
185    let charset = state.charset();
186    let len = charset.len() as f32;
187    let resized = image::imageops::resize(gray, width, height, FilterType::Nearest);
188
189    resized.rows().map(|row| {
190        row.map(|px| {
191            let mut luma = px.0[0] as f32 / 255.0;
192            luma = ((luma - 0.5) * state.contrast + 0.5).clamp(0.0, 1.0);
193            if state.inverted { luma = 1.0 - luma; }
194            let idx = ((luma * (len - 1.0)).round() as usize).min(charset.len() - 1);
195            charset[idx]
196        }).collect()
197    }).collect()
198}

200fn vu_bar(energy: f32) -> String {
201    let n = (energy * 8.0).round() as usize;
202    format!("{}{}", "█".repeat(n), "░".repeat(8usize.saturating_sub(n)))
203}

205// ── Main ──────────────────────────────────────────────────────────────────────
206
207fn main() -> anyhow::Result<()> {
208    // Audio setup
209    let shared_audio: SharedAudio = Arc::new(Mutex::new(BandEnergy::default()));
210    let mut _stream = None;
211    if let Some(dev) = audio::default_device() {
212        eprintln!("Audio: {}", dev.name);
213        _stream = Some(audio::start_capture(dev, shared_audio.clone())?);
214    } else {
215        eprintln!("Kein Audio-Gerät gefunden — laufe ohne Audio");
216    }
217
218    // Webcam
219    let format = RequestedFormat::new::<RgbFormat>(RequestedFormatType::AbsoluteHighestFrameRate);
220    let mut camera = Camera::new(CameraIndex::Index(0), format)?;
221    camera.open_stream()?;
222
223    // Terminal
224    let mut stdout = io::stdout();
225    terminal::enable_raw_mode()?;
226    execute!(stdout, terminal::EnterAlternateScreen, cursor::Hide)?;
227
228    let mut state = State::default();
229    let mut last_script_energy = 0.0f32;
230    let mut running = true;
231
232    while running {
233        // ── Input ─────────────────────────────────────────────────────────
234        if event::poll(Duration::from_millis(0))? {
235            if let Event::Key(KeyEvent { code, .. }) = event::read()? {
236                match code {
237                    KeyCode::Char('q') | KeyCode::Esc => running = false,
238                    KeyCode::Char('i') => state.inverted = !state.inverted,
239                    KeyCode::Char('1') => state.active_band = 0,
240                    KeyCode::Char('2') => state.active_band = 1,
241                    KeyCode::Char('3') => state.active_band = 2,
242                    KeyCode::Char('m') => {
243                        state.mappings[state.active_band] =
244                            state.mappings[state.active_band].cycle();
245                    }
246                    // Manuelle Overrides
247                    KeyCode::Char('+') | KeyCode::Char('=') =>
248                        state.scale = (state.scale + 1).min(8),
249                    KeyCode::Char('-') =>
250                        state.scale = state.scale.saturating_sub(1).max(1),
251                    KeyCode::Char(']') =>
252                        state.contrast = (state.contrast + 0.1).min(4.0),
253                    KeyCode::Char('[') =>
254                        state.contrast = (state.contrast - 0.1).max(0.5),
255                    KeyCode::Char('s') => state.script = state.script.next(),
256                    _ => {}
257                }
258            }
259        }
260
261        // ── Audio ──────────────────────────────────────────────────────────
262        let band = shared_audio.lock().map(|g| g.clone()).unwrap_or_default();
263        state.apply_audio(&band);
264
265        // Script trigger: steigende Flanke über 0.75
266        let script_energy = [band.bass, band.mid, band.high]
267            .iter()
268            .zip(state.mappings.iter())
269            .filter_map(|(&e, &t)| if t == BandTarget::Script { Some(e) } else { None })
270            .fold(0.0f32, f32::max);
271
272        if script_energy > 0.75 && last_script_energy <= 0.75 {
273            state.script = state.script.next();
274        }
275        last_script_energy = script_energy;
276
277        // ── Kamera ─────────────────────────────────────────────────────────
278        let frame = match camera.frame() { Ok(f) => f, Err(_) => continue };
279        let decoded = frame.decode_image::<RgbFormat>()?;
280        let rgb = image::RgbImage::from_raw(decoded.width(), decoded.height(), decoded.into_raw())
281            .expect("buffer mismatch");
282        let gray = image::DynamicImage::ImageRgb8(rgb).to_luma8();
283
284        let (term_cols, term_rows) = terminal::size()?;
285        let scale = state.scale.max(1);
286        let ascii_cols = (term_cols as u32 / scale).max(1);
287        let ascii_rows = (term_rows.saturating_sub(3) as u32 / scale).max(1);
288
289        let lines = frame_to_ascii(&gray, ascii_cols, ascii_rows, &state);
290
291        // ── Render ─────────────────────────────────────────────────────────
292        queue!(stdout, cursor::MoveTo(0, 0), terminal::Clear(ClearType::All))?;
293        for line in &lines {
294            queue!(stdout, Print(line), Print("\r\n"))?;
295        }
296
297        // Band-Anzeige (2. letzte Zeile)
298        let band_data = [
299            ("Bass", band.bass, Color::Red),
300            ("Mid ", band.mid,  Color::Green),
301            ("High", band.high, Color::Cyan),
302        ];
303        queue!(stdout, cursor::MoveTo(0, term_rows - 3))?;
304        for (i, (label, energy, color)) in band_data.iter().enumerate() {
305            let sel = if i == state.active_band { '▶' } else { ' ' };
306            let target = state.mappings[i].name();
307            queue!(
308                stdout,
309                SetForegroundColor(*color),
310                Print(format!("{sel}{label}→{:<9} {} ", target, vu_bar(*energy))),
311                ResetColor,
312            )?;
313        }
314
315        // Status (letzte Zeile)
316        let status = format!(
317            " {script} | kontrast:{:.1} skala:{scale} dichte:{:.0}% | \
318             1/2/3:band  m:remap  i:invert  s:script  [/]:kontrast  +/-:skala  q:quit",
319            state.contrast,
320            state.density * 100.0,
321            script = state.script.name(),
322        );
323        let truncated: String = status.chars().take(term_cols as usize).collect();
324        queue!(stdout, cursor::MoveTo(0, term_rows - 1), Print(truncated))?;
325
326        stdout.flush()?;
327    }
328
329    execute!(stdout, terminal::LeaveAlternateScreen, cursor::Show)?;
330    terminal::disable_raw_mode()?;
331    camera.stop_stream()?;
332    println!("Tschüss!");
333    Ok(())
