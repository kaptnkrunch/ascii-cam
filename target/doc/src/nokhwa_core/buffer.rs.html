26/// A buffer returned by a camera to accommodate custom decoding.
27/// Contains information of Resolution, the buffer's [`FrameFormat`], and the buffer.
28///
29/// Note that decoding on the main thread **will** decrease your performance and lead to dropped frames.
30#[derive(Clone, Debug, Hash, PartialOrd, PartialEq, Eq)]
31pub struct Buffer {
32    resolution: Resolution,
33    buffer: Bytes,
34    source_frame_format: FrameFormat,
35}
36
37impl Buffer {
38    /// Creates a new buffer with a [`&[u8]`].
39    #[must_use]
40    #[inline]
41    pub fn new(res: Resolution, buf: &[u8], source_frame_format: FrameFormat) -> Self {
42        Self {
43            resolution: res,
44            buffer: Bytes::copy_from_slice(buf),
45            source_frame_format,
46        }
47    }
48
49    /// Get the [`Resolution`] of this buffer.
50    #[must_use]
51    pub fn resolution(&self) -> Resolution {
52        self.resolution
53    }
54
55    /// Get the data of this buffer.
56    #[must_use]
57    pub fn buffer(&self) -> &[u8] {
58        &self.buffer
59    }
60
61    /// Get a owned version of this buffer.
62    #[must_use]
63    pub fn buffer_bytes(&self) -> Bytes {
64        self.buffer.clone()
65    }
66
67    /// Get the [`FrameFormat`] of this buffer.
68    #[must_use]
69    pub fn source_frame_format(&self) -> FrameFormat {
70        self.source_frame_format
71    }
72
73    /// Decodes a image with allocation using the provided [`FormatDecoder`].
74    /// # Errors
75    /// Will error when the decoding fails.
76    #[inline]
77    pub fn decode_image<F: FormatDecoder>(
78        &self,
79    ) -> Result<ImageBuffer<F::Output, Vec<u8>>, NokhwaError> {
80        let new_data = F::write_output(self.source_frame_format, self.resolution, &self.buffer)?;
81        let image =
82            ImageBuffer::from_raw(self.resolution.width_x, self.resolution.height_y, new_data)
83                .ok_or(NokhwaError::ProcessFrameError {
84                    src: self.source_frame_format,
85                    destination: stringify!(F).to_string(),
86                    error: "Failed to create buffer".to_string(),
87                })?;
88        Ok(image)
89    }
90
91    /// Decodes a image with allocation using the provided [`FormatDecoder`] into a `buffer`.
92    /// # Errors
93    /// Will error when the decoding fails, or the provided buffer is too small.
94    #[inline]
95    pub fn decode_image_to_buffer<F: FormatDecoder>(
96        &self,
97        buffer: &mut [u8],
98    ) -> Result<(), NokhwaError> {
99        F::write_output_buffer(
100            self.source_frame_format,
101            self.resolution,
102            &self.buffer,
103            buffer,
104        )
105    }

107    /// Decodes an image with allocation using the provided [`FormatDecoder`] into a [`Mat`](https://docs.rs/opencv/latest/opencv/core/struct.Mat.html).
108    ///
109    /// Note that this does a clone when creating the buffer, to decouple the lifetime of the internal data to the temporary Buffer. If you want to avoid this, please see [`decode_opencv_mat`](Self::decode_opencv_mat).
110    ///
111    /// This is **NOT** coherent when the input data is not Gray8, GrayAlpha87, RGB8, or RGBA8
112    /// # Errors
113    /// Will error when the decoding fails, or `OpenCV` failed to create/copy the [`Mat`](https://docs.rs/opencv/latest/opencv/core/struct.Mat.html).
114    /// # Safety
115    /// This function uses `unsafe` in order to create the [`Mat`](https://docs.rs/opencv/latest/opencv/core/struct.Mat.html). Please see [`Mat::new_rows_cols_with_data`](https://docs.rs/opencv/latest/opencv/core/struct.Mat.html#method.new_rows_cols_with_data) for more.
116    ///
117    /// Most notably, the `data` **must** stay in scope for the duration of the [`Mat`](https://docs.rs/opencv/latest/opencv/core/struct.Mat.html) or bad, ***bad*** things happen.
118    #[cfg(feature = "opencv-mat")]
119    #[cfg_attr(feature = "docs-features", doc(cfg(feature = "opencv-mat")))]
120    pub fn decode_opencv_mat<F: FormatDecoder>(
121        &mut self,
122    ) -> Result<BoxedRef<'_, Mat>, NokhwaError> {
123        use crate::buffer::channel_defs::make_mat;
124
125        make_mat::<F>(self.resolution, self.buffer())
126    }

128    /// Decodes an image with allocation using the provided [`FormatDecoder`] into a [`Mat`](https://docs.rs/opencv/latest/opencv/core/struct.Mat.html).
129    ///
130    /// This is **NOT** coherent when the input data is not Gray8, GrayAlpha87, RGB8, or RGBA8
131    /// # Errors
132    /// Will error when the decoding fails, or `OpenCV` failed to create/copy the [`Mat`](https://docs.rs/opencv/latest/opencv/core/struct.Mat.html).
133    #[cfg(feature = "opencv-mat")]
134    #[cfg_attr(feature = "docs-features", doc(cfg(feature = "opencv-mat")))]
135    #[allow(clippy::cast_possible_wrap)]
136    pub fn decode_into_opencv_mat<F: FormatDecoder>(
137        &mut self,
138        dst: &mut Mat,
139    ) -> Result<(), NokhwaError> {
140        use bytes::Buf;
141        use image::Pixel;
142        use opencv::core::{
143            Mat, MatTraitConst, MatTraitManual, Scalar, CV_8UC1, CV_8UC2, CV_8UC3, CV_8UC4,
144        };
145
146        let array_type = match F::Output::CHANNEL_COUNT {
147            1 => CV_8UC1,
148            2 => CV_8UC2,
149            3 => CV_8UC3,
150            4 => CV_8UC4,
151            _ => {
152                return Err(NokhwaError::ProcessFrameError {
153                    src: FrameFormat::RAWRGB,
154                    destination: "OpenCV Mat".to_string(),
155                    error: "Invalid Decoder FormatDecoder Channel Count".to_string(),
156                })
157            }
158        };
159
160        // If destination does not exist, create a new matrix.
161        if dst.empty() {
162            *dst = Mat::new_rows_cols_with_default(
163                self.resolution.height_y as i32,
164                self.resolution.width_x as i32,
165                array_type,
166                Scalar::default(),
167            )
168            .map_err(|why| NokhwaError::ProcessFrameError {
169                src: FrameFormat::RAWRGB,
170                destination: "OpenCV Mat".to_string(),
171                error: why.to_string(),
172            })?;
173        } else {
174            if dst.typ() != array_type {
175                return Err(NokhwaError::ProcessFrameError {
176                    src: FrameFormat::RAWRGB,
177                    destination: "OpenCV Mat".to_string(),
178                    error: "Invalid Matrix Channel Count".to_string(),
179                });
180            }
181
182            if dst.rows() != self.resolution.height_y as _
183                || dst.cols() != self.resolution.width_x as _
184            {
185                return Err(NokhwaError::ProcessFrameError {
186                    src: FrameFormat::RAWRGB,
187                    destination: "OpenCV Mat".to_string(),
188                    error: "Invalid Matrix Dimensions".to_string(),
189                });
190            }
191        }
192
193        let mut bytes = match dst.data_bytes_mut() {
194            Ok(bytes) => bytes,
195            Err(_e) => {
196                return Err(NokhwaError::ProcessFrameError {
197                    src: FrameFormat::RAWRGB,
198                    destination: "OpenCV Mat".to_string(),
199                    error: "Matrix Must Be Continuous".to_string(),
200                })
201            }
202        };
203
204        let mut buffer = self.buffer.as_ref();
205        if bytes.len() != buffer.len() {
206            return Err(NokhwaError::ProcessFrameError {
207                src: FrameFormat::RAWRGB,
208                destination: "OpenCV Mat".to_string(),
209                error: "Matrix Buffer Size Mismatch".to_string(),
210            });
211        }
212
213        buffer.copy_to_slice(&mut bytes);
214
215        Ok(())
216    }
217}

219/// Channel definitions and utilities for making Mat for OpenCV
220///
221/// You (probably) shouldn't use this.
222#[cfg(feature = "opencv-mat")]
223pub mod channel_defs {
224    use crate::error::NokhwaError;
225    use crate::pixel_format::FormatDecoder;
226    use crate::types::{FrameFormat, Resolution};
227    use bytemuck::{cast_slice, Pod, Zeroable};
228    use image::Pixel;
229
230    #[cfg(feature = "opencv-mat")]
231    #[cfg_attr(feature = "docs-features", doc(cfg(feature = "opencv-mat")))]
232    pub(crate) fn make_mat<F>(
233        resolution: Resolution,
234        data: &[u8],
235    ) -> Result<opencv::boxed_ref::BoxedRef<'_, opencv::core::Mat>, NokhwaError>
236    where
237        F: FormatDecoder,
238    {
239        use crate::buffer::channel_defs::*;
240        use opencv::core::Mat;
241
242        let mat = match F::Output::CHANNEL_COUNT {
243            1 => Mat::new_rows_cols_with_data::<G8>(
244                resolution.width() as i32,
245                resolution.height() as i32,
246                cast_slice(data),
247            ),
248            2 => Mat::new_rows_cols_with_data::<GA8>(
249                resolution.width() as i32,
250                resolution.height() as i32,
251                cast_slice(data),
252            ),
253            3 => Mat::new_rows_cols_with_data::<RGB8>(
254                resolution.width() as i32,
255                resolution.height() as i32,
256                cast_slice(data),
257            ),
258            4 => Mat::new_rows_cols_with_data::<RGBA8>(
259                resolution.width() as i32,
260                resolution.height() as i32,
261                cast_slice(data),
262            ),
263            _ => {
264                return Err(NokhwaError::ProcessFrameError {
265                    src: FrameFormat::RAWRGB,
266                    destination: "OpenCV Mat".to_string(),
267                    error: "Invalid Decoder FormatDecoder Channel Count".to_string(),
268                })
269            }
270        };
271
272        match mat {
273            Ok(m) => Ok(m),
274            Err(why) => Err(NokhwaError::ProcessFrameError {
275                src: FrameFormat::RAWRGB,
276                destination: "OpenCV Mat".to_string(),
277                error: why.to_string(),
278            }),
279        }
280    }
