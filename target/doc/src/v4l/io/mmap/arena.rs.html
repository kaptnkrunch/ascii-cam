9/// Manage mapped buffers
10///
11/// All buffers are unmapped in the Drop impl.
12/// In case of errors during unmapping, we panic because there is memory corruption going on.
13pub struct Arena<'a> {
14    handle: Arc<Handle>,
15    pub bufs: Vec<&'a mut [u8]>,
16    pub buf_type: buffer::Type,
17}
18
19impl<'a> Arena<'a> {
20    /// Returns a new buffer manager instance
21    ///
22    /// You usually do not need to use this directly.
23    /// A MappedBufferStream creates its own manager instance by default.
24    ///
25    /// # Arguments
26    ///
27    /// * `handle` - Device handle to get its file descriptor
28    /// * `buf_type` - Type of the buffers
29    pub fn new(handle: Arc<Handle>, buf_type: buffer::Type) -> Self {
30        Arena {
31            handle,
32            bufs: Vec::new(),
33            buf_type,
34        }
35    }
36
37    fn buffer_desc(&self) -> v4l2_buffer {
38        v4l2_buffer {
39            type_: self.buf_type as u32,
40            memory: Memory::Mmap as u32,
41            ..unsafe { mem::zeroed() }
42        }
43    }
44
45    fn requestbuffers_desc(&self) -> v4l2_requestbuffers {
46        v4l2_requestbuffers {
47            type_: self.buf_type as u32,
48            memory: Memory::Mmap as u32,
49            ..unsafe { mem::zeroed() }
50        }
51    }
52
53    pub fn allocate(&mut self, count: u32) -> io::Result<u32> {
54        let mut v4l2_reqbufs = v4l2_requestbuffers {
55            count,
56            ..self.requestbuffers_desc()
57        };
58        unsafe {
59            v4l2::ioctl(
60                self.handle.fd(),
61                v4l2::vidioc::VIDIOC_REQBUFS,
62                &mut v4l2_reqbufs as *mut _ as *mut std::os::raw::c_void,
63            )?;
64        }
65
66        for index in 0..v4l2_reqbufs.count {
67            let mut v4l2_buf = v4l2_buffer {
68                index,
69                ..self.buffer_desc()
70            };
71            unsafe {
72                v4l2::ioctl(
73                    self.handle.fd(),
74                    v4l2::vidioc::VIDIOC_QUERYBUF,
75                    &mut v4l2_buf as *mut _ as *mut std::os::raw::c_void,
76                )?;
77
78                let ptr = v4l2::mmap(
79                    ptr::null_mut(),
80                    v4l2_buf.length as usize,
81                    libc::PROT_READ | libc::PROT_WRITE,
82                    libc::MAP_SHARED,
83                    self.handle.fd(),
84                    v4l2_buf.m.offset as libc::off_t,
85                )?;
86
87                let slice =
88                    slice::from_raw_parts_mut::<u8>(ptr as *mut u8, v4l2_buf.length as usize);
89                self.bufs.push(slice);
90            }
91        }
92
93        Ok(v4l2_reqbufs.count)
94    }
95
96    pub fn release(&mut self) -> io::Result<()> {
97        for buf in &self.bufs {
98            unsafe {
99                v4l2::munmap(buf.as_ptr() as *mut core::ffi::c_void, buf.len())?;
100            }
101        }
102
103        // free all buffers by requesting 0
104        let mut v4l2_reqbufs = v4l2_requestbuffers {
105            count: 0,
106            ..self.requestbuffers_desc()
107        };
108        unsafe {
109            v4l2::ioctl(
110                self.handle.fd(),
111                v4l2::vidioc::VIDIOC_REQBUFS,
112                &mut v4l2_reqbufs as *mut _ as *mut std::os::raw::c_void,
113            )?;
114        }
115
116        self.bufs.clear();
117        Ok(())
118    }
119}
120
121impl<'a> Drop for Arena<'a> {
122    fn drop(&mut self) {
123        if self.bufs.is_empty() {
124            // nothing to do
125            return;
126        }
127
128        if let Err(e) = self.release() {
129            if let Some(code) = e.raw_os_error() {
130                // ENODEV means the file descriptor wrapped in the handle became invalid, most
131                // likely because the device was unplugged or the connection (USB, PCI, ..)
132                // broke down. Handle this case gracefully by ignoring it.
133                if code == 19 {
134                    /* ignore */
135                    return;
136                }
137            }
138
139            panic!("{:?}", e)
140        }
141    }
142}
